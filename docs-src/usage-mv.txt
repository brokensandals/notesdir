usage: notesdir mv [-h] [-c] [-j] src dest

positional arguments:
  src                   file or folder to move
  dest                  new filename or new parent folder

optional arguments:
  -h, --help            show this help message and exit
  -c, --creation-folders
                        insert folders like 2020/06 based on creation date of
                        src
  -j, --json            output as json
